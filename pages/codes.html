#include <stdio.h>

int main() {
    int a, b;
    
    printf("What is the length and width of your rectangle?\n");
    printf("length:\t");
    scanf("%d", &a);
    printf("width:\t");
    scanf("%d", &b);
    printf("The perimeter of your rectangle is %d \n", (2*a)+(2*b));
    printf("The area of your rectangle is %d", (2*a)*(2*b));
    return 0;
}

    const double pi = 3.14;

    printf("The circumference of your circle is %.2lf \n", (2*pi)*a);

    return 0;
}
